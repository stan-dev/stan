579  &lp))
580  return lp;
581  if (!check_size_match(function,
582  y.cols(), "Columns of random variable",
583  Sigma.rows(), "rows of covariance parameter",
584  &lp))
585  return lp;
586  if (!check_size_match(function,
587  y.cols(), "Columns of random variable",
588  Sigma.cols(), "columns of covariance parameter",
589  &lp))
590  return lp;
591  if (!check_finite(function, mu, "Location parameter", &lp))
592  return lp;
593  if (!check_not_nan(function, y, "Random variable", &lp))
594  return lp;
595 
596  if (y.cols() == 0)
597  return lp;
598 
599  if (include_summand<propto>::value)
600  lp += NEG_LOG_SQRT_TWO_PI * y.cols() * y.rows();
601 
602  if (include_summand<propto,T_covar>::value) {
603  lp += log_determinant_ldlt(ldlt_Sigma) * (0.5 * y.cols());
604  }
605 
606  if (include_summand<propto,T_y,T_loc,T_covar>::value) {
607  Eigen::Matrix<T_loc, Eigen::Dynamic, Eigen::Dynamic> MU(y.rows(),y.cols());
608  for (typename Eigen::Matrix<T_loc, Eigen::Dynamic, Eigen::Dynamic>::size_type i = 0;
609  i < y.rows(); i++)
610  MU.row(i) = mu;
611 
612  Eigen::Matrix<typename
613  boost::math::tools::promote_args<T_loc,T_y>::type,
614  Eigen::Dynamic,Eigen::Dynamic> y_minus_MU(y.rows(), y.cols());
615 
616  for (int i = 0; i < y.size(); i++)
617  y_minus_MU(i) = y(i)-MU(i);
618 
619  Eigen::Matrix<typename
620  boost::math::tools::promote_args<T_loc,T_y>::type,
621  Eigen::Dynamic,Eigen::Dynamic> z(y_minus_MU.transpose()); // was =
622 
623  lp -= 0.5 * trace_quad_form(Sigma,z);
624  }
625  return lp;
626  }
