253  rho_hat_t.setZero();

258  for (int chain = 0; chain < chains; chain++) {
259  acov_t(chain) = acov(chain)(t);
260  }
261  rho_hat = 1 - (mean_var - mean(acov_t)) / var_plus;
262  if (rho_hat >= 0)

264  max_t = t;
265  }
