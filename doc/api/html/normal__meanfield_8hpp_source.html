167  // 0.5 * dim * (1+log2pi) + 0.5 * log det diag(sigma^2) =
168  // 0.5 * dim * (1+log2pi) + sum(log(sigma)) =
169  // 0.5 * dim * (1+log2pi) + sum(omega)
170  double entropy() const {
171  return 0.5 * static_cast<double>(dimension_) *
172  (1.0 + stan::math::LOG_TWO_PI) + omega_.sum();
173  }

238  // Initialize everything to zero
239  Eigen::VectorXd mu_grad = Eigen::VectorXd::Zero(dimension_);
240  Eigen::VectorXd omega_grad = Eigen::VectorXd::Zero(dimension_);
241  double tmp_lp = 0.0;
242  Eigen::VectorXd tmp_mu_grad = Eigen::VectorXd::Zero(dimension_);
243  Eigen::VectorXd eta = Eigen::VectorXd::Zero(dimension_);
244  Eigen::VectorXd zeta = Eigen::VectorXd::Zero(dimension_);
245 
246  // Naive Monte Carlo integration
247  for (int i = 0; i < n_monte_carlo_grad; ++i) {
248  // Draw from standard normal and transform to real-coordinate space
249  for (int d = 0; d < dimension_; ++d) {
250  eta(d) = stan::math::normal_rng(0, 1, rng);
251  }
252  zeta = transform(eta);
253 
254  stan::math::check_not_nan(function, "zeta", zeta);
255 
256  // Compute gradient step in real-coordinate space
257  stan::model::gradient(m, zeta, tmp_lp, tmp_mu_grad, print_stream);
258 
259  // Update gradient parameters
260  mu_grad += tmp_mu_grad;
261  omega_grad.array() += tmp_mu_grad.array().cwiseProduct(eta.array());
262  }
263  mu_grad /= static_cast<double>(n_monte_carlo_grad);
264  omega_grad /= static_cast<double>(n_monte_carlo_grad);
265 
266  // Multiply by exp(omega)
267  omega_grad.array() =
268  omega_grad.array().cwiseProduct(omega_.array().exp());
269 
270  // Add gradient of entropy term (just equal to element-wise 1 here)
271  omega_grad.array() += 1.0;
272
