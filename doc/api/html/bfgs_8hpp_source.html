157  int step() {
158  Scalar gradNorm, stepNorm;
159  VectorT sk, yk;
160  int retCode(0);
161  int resetB(0);
162 
163  _itNum++;
164 
165  if (_itNum == 1) {
166  resetB = 1;
167  _note = "";
168  }
169  else {
170  resetB = 0;
171  _note = "";
172  }
173 
174  while (true) {
175  if (resetB) {
176  // Reset the Hessian approximation
177  _pk.noalias() = -_gk;
178  }
179 
180  // Get an initial guess for the step size (alpha)
181  if (_itNum > 1 && resetB != 2) {
182  // use cubic interpolation based on the previous step
183  _alpha0 = _alpha = std::min(1.0,
184  1.01*CubicInterp(_gk_1.dot(_pk_1),
185  _alphak_1,
186  _fk - _fk_1,
187  _gk.dot(_pk_1),
188  _ls_opts.minAlpha, 1.0));
189  }
190  else {
191  // On the first step (or, after a reset) use the default step size
192  _alpha0 = _alpha = _ls_opts.alpha0;
193  }
194 
195  // Perform the line search. If successful, the results are in the
196  // variables: _xk_1, _fk_1 and _gk_1.
197  retCode = WolfeLineSearch(_func, _alpha, _xk_1, _fk_1, _gk_1,
198  _pk, _xk, _fk, _gk,
199  _ls_opts.c1, _ls_opts.c2,
200  _ls_opts.minAlpha);
201  if (retCode) {
202  // Line search failed...
203  if (resetB) {
204  // did a Hessian reset and it still failed, and nothing left to try
205  retCode = TERM_LSFAIL;
206  return retCode;
207  }
208  else {
209  // try resetting the Hessian approximation
210  resetB = 2;
211  _note += "LS failed, Hessian reset";
212  continue;
213  }
214  }
215  else {
216  break;
217  }
218  }
219 
220  // Swap things so that k is the most recent iterate
221  std::swap(_fk,_fk_1);
222  _xk.swap(_xk_1);
223  _gk.swap(_gk_1);
224  _pk.swap(_pk_1);
225 
226  sk.noalias() = _xk - _xk_1;
227  yk.noalias() = _gk - _gk_1;
228 
229  gradNorm = _gk.norm();
230  stepNorm = sk.norm();
231 
232  // Update QN approximation
233  if (resetB) {
234  // If the QN approximation was reset, automatically scale it
235  // and update the step-size accordingly
236  Scalar B0fact = _qn.update(yk,sk,true);
237  _pk_1 /= B0fact;
238  _alphak_1 = _alpha*B0fact;
239  }
240  else {
241  _qn.update(yk,sk);
242  _alphak_1 = _alpha;
243  }
244 
245 
246  // Compute search direction for next step
247  _qn.search_direction(_pk,_gk);
248 
249  // Check for convergence
250  if (std::fabs(_fk_1 - _fk) < _conv_opts.tolAbsF) {
251  retCode = TERM_ABSF; // Objective function improvement wasn't sufficient
252  }
253  else if (gradNorm < _conv_opts.tolAbsGrad) {
254  retCode = TERM_ABSGRAD; // Gradient norm was below threshold
255  }
256  else if (stepNorm < _conv_opts.tolAbsX) {
257  retCode = TERM_ABSX; // Change in x was too small
258  }
259  else if (_itNum >= _conv_opts.maxIts) {
260  retCode = TERM_MAXIT; // Max number of iterations hit
261  }
262  else if (rel_obj_decrease() < _conv_opts.tolRelF*std::numeric_limits<Scalar>::epsilon()) {
263  retCode = TERM_RELF; // Relative improvement in objective function wasn't sufficient
264  }
265  else if (rel_grad_norm() < _conv_opts.tolRelGrad*std::numeric_limits<Scalar>::epsilon()) {
266  retCode = TERM_RELGRAD; // Relative gradient norm was below threshold
267  }
268  else {
269  retCode = TERM_SUCCESS; // Step was successful more progress to be made
270  }
271 
272  return retCode;
273  }
