175  // 0.5 * dim * (1+log2pi) + 0.5 * log det (L^T L) =
176  // 0.5 * dim * (1+log2pi) + sum(log(abs(diag(L))))
177  double entropy() const {
178  double tmp = 0.0;
179  double result = 0.5 * static_cast<double>(dimension_) *
180  (1.0 + stan::math::LOG_TWO_PI);
181  for (int d = 0; d < dimension_; ++d) {
182  tmp = fabs(L_chol_(d, d));
183  if (tmp != 0.0) {
184  result += log(tmp);
185  }
186  }
187  return result;
188  }

251 
252  // Initialize everything to zero
253  Eigen::VectorXd mu_grad = Eigen::VectorXd::Zero(dimension_);
254  Eigen::MatrixXd L_grad = Eigen::MatrixXd::Zero(dimension_, dimension_);
255  double tmp_lp = 0.0;
256  Eigen::VectorXd tmp_mu_grad = Eigen::VectorXd::Zero(dimension_);
257  Eigen::VectorXd eta = Eigen::VectorXd::Zero(dimension_);
258  Eigen::VectorXd zeta = Eigen::VectorXd::Zero(dimension_);
259 
260  // Naive Monte Carlo integration
261  for (int i = 0; i < n_monte_carlo_grad; ++i) {
262  // Draw from standard normal and transform to real-coordinate space
263  for (int d = 0; d < dimension_; ++d) {
264  eta(d) = stan::math::normal_rng(0, 1, rng);
265  }
266  zeta = transform(eta);
267 
268  // Compute gradient step in real-coordinate space
269  stan::model::gradient(m, zeta, tmp_lp, tmp_mu_grad, print_stream);
270 
271  // Update gradient parameters
272  mu_grad += tmp_mu_grad;
273  for (int ii = 0; ii < dimension_; ++ii) {
274  for (int jj = 0; jj <= ii; ++jj) {
275  L_grad(ii, jj) += tmp_mu_grad(ii) * eta(jj);
276  }
277  }
278  }
279  mu_grad /= static_cast<double>(n_monte_carlo_grad);
280  L_grad /= static_cast<double>(n_monte_carlo_grad);
281 
282  // Add gradient of entropy term
283  L_grad.diagonal().array() += L_chol_.diagonal().array().inverse();
284 
285  // Set parameters to argument
286  elbo_grad.set_mu(mu_grad);
287  elbo_grad.set_L_chol(L_grad);
