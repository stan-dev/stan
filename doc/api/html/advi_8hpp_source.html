96  double calc_ELBO(const Q& variational) const {
97  double elbo = 0.0;
98  int dim = variational.dimension();
99 
100  Eigen::VectorXd zeta = Eigen::VectorXd::Zero(dim);
101 
102  for (int i = 0; i < n_monte_carlo_elbo_; ++i) {
103  // Draw from variational distribution
104  zeta = variational.sample(rng_);
105 
106  // Accumulate log probability
107  elbo += (model_.template log_prob<false, true>(zeta, print_stream_));
108  }
109 
110  // Divide to get Monte Carlo integral estimate
111  elbo /= n_monte_carlo_elbo_;
112 
113  // Add entropy term
114  elbo += variational.entropy();
115 
116  return elbo;
117  }

159  max_iterations);
160 
161  // Gradient parameters
162  Q elbo_grad = Q(model_.num_params_r());
163 
164  // Adagrad parameters
165  double tau = 1.0;
166  Q params_adagrad = Q(model_.num_params_r());
167 
168  // RMSprop window_size
169  double window_size = 10.0;
170  double post_factor = 1.0 / window_size;
171  double pre_factor = 1.0 - post_factor;
172 
173  // Initialize ELBO and convergence tracking variables
174  double elbo(0.0);
175  double elbo_prev = std::numeric_limits<double>::min();
176  double delta_elbo = std::numeric_limits<double>::max();
177  double delta_elbo_ave = std::numeric_limits<double>::max();
178  double delta_elbo_med = std::numeric_limits<double>::max();
179 
180  // Heuristic to estimate how far to look back in rolling window
181  int cb_size = static_cast<int>(
182  std::max(0.1*max_iterations/static_cast<double>(eval_elbo_),
183  1.0));
184  boost::circular_buffer<double> cb(cb_size);
185 
186  // Print stuff
187  if (print_stream_) {
188  *print_stream_ << " iter"
189  << " ELBO"

191  << " delta_ELBO_med"
192  << " notes "
193  << std::endl;
194  }
195 
196  // Timing variables
197  clock_t start = clock();
198  clock_t end;
199  double delta_t;
200 
201  // Main loop
202  std::vector<double> print_vector;
203  bool do_more_iterations = true;
204  int iter_counter = 0;
205  while (do_more_iterations) {
206  // Compute gradient using Monte Carlo integration
207  calc_ELBO_grad(variational, elbo_grad);
208 
209  // RMSprop moving average weighting
210  if (iter_counter == 0) {
211  params_adagrad += elbo_grad.square();
212  } else {
213  params_adagrad = pre_factor * params_adagrad +
214  post_factor * elbo_grad.square();
215  }
216 
217  // Stochastic gradient update
218  variational += eta_adagrad_ * elbo_grad /
219  (tau + params_adagrad.sqrt());
220 
221  // Check for convergence every "eval_elbo_"th iteration
222  if (iter_counter % eval_elbo_ == 0) {
223  elbo_prev = elbo;
224  elbo = calc_ELBO(variational);
225  delta_elbo = rel_decrease(elbo, elbo_prev);
226  cb.push_back(delta_elbo);
227  delta_elbo_ave = std::accumulate(cb.begin(), cb.end(), 0.0)
228  / static_cast<double>(cb.size());
229  delta_elbo_med = circ_buff_median(cb);

242  << delta_elbo_med;
243  }

248
